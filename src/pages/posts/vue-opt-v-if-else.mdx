## Simple example using the v-if directive

```html
<body>
        
<div id="app">
    <p v-if="characterList.length === 0">There are no characters</p>
    <ul v-else-if = "characterList.length % 2 === 0">
        <li v-for="(character, index) in characterList">{{ index }}{{ character.name}}</li>
    </ul>
    <p v-else>There is an odd number of characters</p>
    </div>  

    <script src="https://unpkg.com/vue@3/dist/vue.global.js"></script>
<script>
    Vue.createApp({
        data: () => ({
            characterList: [
                {name: "james"},
                {name: "david"},
                {name: "paul"}
            ]
        }) j
    }).mount('#app')
</script>
</body>
```
